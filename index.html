""" Clap Jump - 拍手跳躍遊戲 (Python 版) ===================================== 需要安裝的套件： pip install pygame opencv-python mediapipe 操作方式： - 雙手靠近（拍手）→ 跳躍 - 躲避從右邊來的紅色障礙物 - 按 ESC 或關閉視窗結束遊戲 """ import sys import math import random import threading import time import cv2 import mediapipe as mp import pygame # ────────────────────────────────────────────── # 設定常數 # ────────────────────────────────────────────── WINDOW_W, WINDOW_H = 960, 540 FPS = 60 GRAVITY = 0.55 GROUND_Y = WINDOW_H - 120 # 地面 Y 座標（角色底部） DINO_W = 50 DINO_H = 50 OBSTACLE_W = 40 OBSTACLE_H = 60 JUMP_VELOCITY = -16 JUMP_COOLDOWN = 0.25 # 秒 # 相機設定：設為 None 讓程式自動偵測前置鏡頭； # 若自動選錯，請改成 0, 1, 2… 手動指定。 CAMERA_INDEX = None # 顏色 C_BG_TOP = (26, 26, 46) C_BG_BOT = (22, 33, 62) C_GREEN = (92, 251, 108) C_RED = (255, 80, 80) C_GROUND = (68, 68, 68) C_WHITE = (230, 230, 230) C_YELLOW = (255, 220, 50) C_DARK = (20, 20, 20) C_PANEL = (0, 0, 0, 180) # RGBA # ────────────────────────────────────────────── # MediaPipe 手部偵測執行緒 # ────────────────────────────────────────────── class HandDetector: def __init__(self): self.mp_hands = mp.solutions.hands self.hands = self.mp_hands.Hands( max_num_hands=2, min_detection_confidence=0.5, min_tracking_confidence=0.5, model_complexity=0 ) self.cap = self._open_front_camera() self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640) self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480) self._lock = threading.Lock() self._clap = False # 本幀是否偵測到拍手 self._hand_dist = 0.0 self._hand_vel = 0.0 self._num_hands = 0 self._last_dist = 0.0 self._last_time = time.time() self._cam_frame = None # 最新相機畫面（BGR） self._running = True self._thread = threading.Thread(target=self._loop, daemon=True) self._thread.start() # ── 自動尋找前鏡頭（自拍鏡頭）── @staticmethod def _open_front_camera(): """ 依序嘗試找到前置鏡頭（自拍用）。 策略： 1. 先列舉所有可用的 index (0~9)，收集能開啟的裝置。 2. 若有多個，從裝置名稱關鍵字判斷前置鏡頭 (front / facetime / selfie / isight / built-in / 內建)。 3. 找不到關鍵字就回退到 index 0（通常已是前置）。 macOS 上 AVFoundation backend (cv2.CAP_AVFOUNDATION) 能 直接讀到 FaceTime HD 鏡頭；Windows / Linux 則用預設 backend。 """ import platform # 使用者手動指定鏡頭 if CAMERA_INDEX is not None: print(f"[Camera] 使用手動指定 index={CAMERA_INDEX}") return cv2.VideoCapture(CAMERA_INDEX) backend = cv2.CAP_AVFOUNDATION if platform.system() == "Darwin" else cv2.CAP_ANY # 收集可用鏡頭 available = [] for idx in range(10): cap = cv2.VideoCapture(idx, backend) if cap.isOpened(): # 嘗試讀一幀確認真的有畫面 ok, _ = cap.read() if ok: available.append((idx, cap)) continue cap.release() if not available: print("[Camera] 找不到任何相機，嘗試強制開啟 index 0…") return cv2.VideoCapture(0) if len(available) == 1: print(f"[Camera] 只找到一個相機 index={available[0][0]}，直接使用。") # 釋放其餘（這裡只有一個，不需多做） return available[0][1] # 有多個鏡頭 → 用關鍵字猜前置 FRONT_KEYWORDS = ("front", "facetime", "selfie", "isight", "built-in", "builtin", "internal", "内建", "內建") front_idx, front_cap = available[0][0], available[0][1] # 預設第一個 for idx, cap in available: # cv2 沒有直接查詢裝置名稱的 API， # 但在某些平台可透過 backend_name 或 fourcc 推測； # 這裡改用「index 越小越可能是前置」的啟發式規則， # 並印出提示讓使用者知道選了哪顆。 print(f"[Camera] 可用鏡頭 index={idx}") # 釋放未選中的 for idx, cap in available: if idx != front_idx: cap.release() print(f"[Camera] 使用 index={front_idx}（若想換鏡頭請修改程式頂端 CAMERA_INDEX 常數）") return front_cap # ── 內部偵測迴圈（背景執行緒） ── def _loop(self): while self._running: ok, frame = self.cap.read() if not ok: time.sleep(0.01) continue frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB) results = self.hands.process(frame_rgb) clap = False dist = 0.0 velocity = 0.0 n_hands = 0 if results.multi_hand_landmarks: n_hands = len(results.multi_hand_landmarks) if n_hands >= 2: def center(lms): xs = [lm.x for lm in lms.landmark] ys = [lm.y for lm in lms.landmark] return (sum(xs)/len(xs), sum(ys)/len(ys)) c1 = center(results.multi_hand_landmarks[0]) c2 = center(results.multi_hand_landmarks[1]) # 使用正規化座標（0~1），乘以畫面尺寸 dx = (c1[0] - c2[0]) * 640 dy = (c1[1] - c2[1]) * 480 dist = math.sqrt(dx*dx + dy*dy) now = time.time() dt = max(now - self._last_time, 0.001) velocity = (self._last_dist - dist) / dt # px/s；正值 = 靠近 self._last_dist = dist self._last_time = now clap = velocity > 200 # 閾值：靠近速度 > 200 px/s with self._lock: self._clap = clap self._hand_dist = dist self._hand_vel = velocity self._num_hands = n_hands self._cam_frame = cv2.flip(frame, 1) # 水平翻轉（鏡像） # ── 供主迴圈讀取 ── def get_state(self): """回傳 (clap_triggered, hand_dist, hand_vel, num_hands)""" with self._lock: state = (self._clap, self._hand_dist, self._hand_vel, self._num_hands) self._clap = False # 消耗一次拍手事件 return state def get_frame(self): with self._lock: return self._cam_frame def stop(self): self._running = False self.cap.release() # ────────────────────────────────────────────── # 障礙物 # ────────────────────────────────────────────── class Obstacle: def __init__(self, speed): # 隨機高度變化讓遊戲更多樣 self.h = random.randint(40, 80) self.w = OBSTACLE_W self.x = float(WINDOW_W + 20) self.y = float(GROUND_Y - self.h + DINO_H) self.speed = speed def update(self): self.x -= self.speed def draw(self, screen): rect = pygame.Rect(int(self.x), int(self.y), self.w, self.h) pygame.draw.rect(screen, C_RED, rect, border_radius=4) pygame.draw.rect(screen, (200, 40, 40), rect, width=2, border_radius=4) def is_off_screen(self): return self.x + self.w < 0 def collides(self, dx, dy, dw, dh): """AABB 碰撞（略微縮小碰撞框增加容錯）""" margin = 8 return not ( dx + dw - margin < self.x + margin or dx + margin > self.x + self.w - margin or dy + dh - margin < self.y + margin or dy + margin > self.y + self.h - margin ) # ────────────────────────────────────────────── # 主遊戲 # ────────────────────────────────────────────── class Game: def __init__(self): pygame.init() pygame.display.set_caption("Clap Jump - 拍手跳躍遊戲") self.screen = pygame.display.set_mode((WINDOW_W, WINDOW_H)) self.clock = pygame.time.Clock() # 字型 pygame.font.init() self.font_big = pygame.font.SysFont("Arial", 48, bold=True) self.font_med = pygame.font.SysFont("Arial", 28, bold=True) self.font_small = pygame.font.SysFont("Arial", 20) # 背景漸層 self._bg = self._make_gradient() self.detector = HandDetector() self._reset() def _make_gradient(self): surf = pygame.Surface((WINDOW_W, WINDOW_H)) for y in range(WINDOW_H): t = y / WINDOW_H r = int(C_BG_TOP[0] + (C_BG_BOT[0]-C_BG_TOP[0])*t) g = int(C_BG_TOP[1] + (C_BG_BOT[1]-C_BG_TOP[1])*t) b = int(C_BG_TOP[2] + (C_BG_BOT[2]-C_BG_TOP[2])*t) pygame.draw.line(surf, (r,g,b), (0,y), (WINDOW_W,y)) return surf def _reset(self): self.score = 0 self.game_speed = 5.0 # 基礎障礙物移動速度（px/frame） self.running = True self.game_over = False self.obstacles = [] self.spawn_counter = 0 self.spawn_rate = 150 # frames self.dino_x = 80.0 self.dino_y = float(GROUND_Y) self.dino_vy = 0.0 self.is_jumping = False self.last_jump_time = 0.0 self.particle_list = [] # 跳躍粒子特效 # ── 跳躍 ── def _jump(self): now = time.time() if not self.is_jumping and now - self.last_jump_time > JUMP_COOLDOWN: self.dino_vy = JUMP_VELOCITY self.is_jumping = True self.last_jump_time = now # 粒子特效 for _ in range(8): self.particle_list.append({ 'x': self.dino_x + DINO_W/2, 'y': self.dino_y + DINO_H, 'vx': random.uniform(-3, 3), 'vy': random.uniform(1, 4), 'life': 20 }) # ── 更新物理 ── def _update_dino(self): self.dino_vy += GRAVITY self.dino_y += self.dino_vy if self.dino_y >= GROUND_Y: self.dino_y = GROUND_Y self.dino_vy = 0.0 self.is_jumping = False def _update_particles(self): for p in self.particle_list: p['x'] += p['vx'] p['y'] += p['vy'] p['vy'] += 0.3 p['life'] -= 1 self.particle_list = [p for p in self.particle_list if p['life'] > 0] # ── 繪製 ── def _draw_dino(self): x, y = int(self.dino_x), int(self.dino_y) # 身體 pygame.draw.rect(self.screen, C_GREEN, (x, y, DINO_W, DINO_H), border_radius=6) pygame.draw.rect(self.screen, (0, 200, 0), (x, y, DINO_W, DINO_H), width=3, border_radius=6) # 眼睛 pygame.draw.circle(self.screen, C_DARK, (x+15, y+15), 5) pygame.draw.circle(self.screen, C_DARK, (x+32, y+15), 5) pygame.draw.circle(self.screen, C_WHITE, (x+16, y+13), 2) pygame.draw.circle(self.screen, C_WHITE, (x+33, y+13), 2) # 粒子 for p in self.particle_list: alpha = int(255 * p['life'] / 20) color = (92, 251, 108, alpha) surf = pygame.Surface((6,6), pygame.SRCALPHA) pygame.draw.circle(surf, color, (3,3), 3) self.screen.blit(surf, (int(p['x'])-3, int(p['y'])-3)) def _draw_hud(self, clap, dist, vel, n_hands): # 左上角資訊面板 panel = pygame.Surface((220, 90), pygame.SRCALPHA) panel.fill((0,0,0,160)) self.screen.blit(panel, (10, 10)) score_txt = self.font_med.render(f"得分: {self.score}", True, C_GREEN) speed_txt = self.font_small.render(f"速度: {self.game_speed/5:.1f}x", True, C_WHITE) self.screen.blit(score_txt, (20, 16)) self.screen.blit(speed_txt, (20, 50)) # 底部偵測狀態列 if n_hands >= 2: color = C_GREEN if clap else C_WHITE status = f"👋 距離: {dist:.0f}px | 速度: {vel:.0f}px/s" else: color = (180, 180, 180) status = f"👋 手部偵測: 需要 2 隻手（目前偵測到 {n_hands} 隻）" det_panel = pygame.Surface((WINDOW_W - 20, 30), pygame.SRCALPHA) det_panel.fill((0,0,0,140)) self.screen.blit(det_panel, (10, WINDOW_H - 40)) det_txt = self.font_small.render(status, True, color) self.screen.blit(det_txt, (20, WINDOW_H - 34)) def _draw_game_over(self): # 半透明遮罩 overlay = pygame.Surface((WINDOW_W, WINDOW_H), pygame.SRCALPHA) overlay.fill((0,0,0,160)) self.screen.blit(overlay, (0,0)) panel_w, panel_h = 400, 220 px, py = (WINDOW_W-panel_w)//2, (WINDOW_H-panel_h)//2 panel = pygame.Surface((panel_w, panel_h), pygame.SRCALPHA) panel.fill((20,20,20,230)) self.screen.blit(panel, (px, py)) pygame.draw.rect(self.screen, C_RED, (px,py,panel_w,panel_h), width=3, border_radius=10) over_txt = self.font_big.render("遊戲結束！", True, C_RED) score_txt = self.font_med.render(f"最終得分：{self.score}", True, C_WHITE) hint_txt = self.font_small.render("按 R 重新開始 | 按 ESC 離開", True, (180,180,180)) self.screen.blit(over_txt, (px + (panel_w - over_txt.get_width())//2, py+30)) self.screen.blit(score_txt, (px + (panel_w - score_txt.get_width())//2, py+100)) self.screen.blit(hint_txt, (px + (panel_w - hint_txt.get_width())//2, py+160)) def _draw_cam_preview(self, frame): """右上角顯示相機畫面（小視窗）""" if frame is None: return h, w = 90, 160 frame_small = cv2.resize(frame, (w, h)) frame_rgb = cv2.cvtColor(frame_small, cv2.COLOR_BGR2RGB) surf = pygame.surfarray.make_surface(frame_rgb.swapaxes(0,1)) self.screen.blit(surf, (WINDOW_W - w - 10, 10)) pygame.draw.rect(self.screen, C_WHITE, (WINDOW_W-w-10, 10, w, h), width=2) def _draw_start_hint(self): txt = self.font_small.render("準備好後請拍手開始！讓 MediaPipe 偵測到雙手…", True, C_YELLOW) self.screen.blit(txt, ((WINDOW_W - txt.get_width())//2, WINDOW_H//2 - 20)) # ── 主迴圈 ── def run(self): while self.running: # ── 事件 ── for event in pygame.event.get(): if event.type == pygame.QUIT: self.running = False elif event.type == pygame.KEYDOWN: if event.key == pygame.K_ESCAPE: self.running = False elif event.key == pygame.K_r and self.game_over: self._reset() elif event.key == pygame.K_SPACE and not self.game_over: # 空白鍵也可以跳（測試用） self._jump() # ── 手部偵測 ── clap, dist, vel, n_hands = self.detector.get_state() cam_frame = self.detector.get_frame() # ── 遊戲邏輯 ── if not self.game_over: if clap: self._jump() self._update_dino() self._update_particles() # 生成障礙物 self.spawn_counter += 1 if self.spawn_counter >= self.spawn_rate: self.obstacles.append(Obstacle(self.game_speed)) self.spawn_counter = 0 self.spawn_rate = max(60, 150 - self.score * 2) # 更新 / 移除障礙物 for obs in self.obstacles[:]: obs.update() if obs.is_off_screen(): self.obstacles.remove(obs) self.score += 1 # 每 5 分加速 if self.score % 5 == 0: self.game_speed += 0.5 # 碰撞偵測 for obs in self.obstacles: if obs.collides(self.dino_x, self.dino_y, DINO_W, DINO_H): self.game_over = True break # ──────────────────── 繪製 ──────────────────── self.screen.blit(self._bg, (0,0)) # 地面線 pygame.draw.line(self.screen, C_GROUND, (0, GROUND_Y + DINO_H), (WINDOW_W, GROUND_Y + DINO_H), 2) # 地面底紋 for gx in range(0, WINDOW_W, 40): pygame.draw.line(self.screen, (50,50,60), (gx, GROUND_Y + DINO_H), (gx, WINDOW_H), 1) # 障礙物 for obs in self.obstacles: obs.draw(self.screen) # 角色 self._draw_dino() # HUD self._draw_hud(clap, dist, vel, n_hands) # 相機小視窗 self._draw_cam_preview(cam_frame) if self.game_over: self._draw_game_over() elif n_hands < 2: self._draw_start_hint() pygame.display.flip() self.clock.tick(FPS) # 清理 self.detector.stop() pygame.quit() sys.exit() # ────────────────────────────────────────────── if __name__ == "__main__": game = Game() game.run()
